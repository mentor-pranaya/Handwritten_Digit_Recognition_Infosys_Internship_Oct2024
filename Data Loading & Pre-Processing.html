def vis_dt(images, labels, title = 'Original Images'):
    fig, axes = plt.subplots(2, 5, figsize=(10, 4))
    for i in range(10):
        ax = axes[i//5, i% 5]
        ax.imshow(images[i].squeeze().numpy(), cmap='gray')
        ax.set_title(f'Original: {labels[i].item()}')
        ax.axis('off')
plt.tight_layout()
plt.show()
